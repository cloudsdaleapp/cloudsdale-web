def pid_alive?(pid)
  begin
    ::Process.kill(0, pid)
    true
  rescue Errno::ESRCH
    false
  end
end

def safefork
  tryagain = true  
  while tryagain
    tryagain = false
    begin
      if pid = fork
        return pid
      end
    rescue Errno::EWOULDBLOCK
      sleep 5
      tryagain = true
    end
  end
end
